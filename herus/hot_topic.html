def query_4db(query):
    df = pd.DataFrame()
    
    for db in db_list:
        
        params['database'] = db
        conn = psycopg2.connect(**params)
        print(">>> Query database {}".format(db))

        df_query = pd.read_sql_query(query, conn)
        df = pd.concat([df, df_query])
    
    return df

def plot_stats(topic):
    """
    Given the name of a topic, plots :
        - Volume over time
        - Histogram of tweets by users
        - Users who twitted the most
    /!\ Local params : data, topic_dict
    """
    selector = data.body.str.contains('(?i)'+'|'.join(topic_dict[topic]))
    data[selector].groupby('publishing_month_year')['publishing_month_year'].count()\
        .iplot(kind='bar',
               title="Tweets about '{}' over time".format(topic))

    # Histogram of tweets. How much do people tweet
    """
    data[selector].groupby('author_id')['author_id'].count().iplot(kind='hist',
                                                                   title='Histogram of the # tweets')
    """
    gb = data[selector].groupby('author_id')['author_id'].count()
    fig_data = go.Histogram(x=gb.values)
    layout = go.Layout(
        title = "Histogram of tweets by user",
        xaxis = dict(title='# of Tweets'),
        yaxis = dict(title='# of Tweeters', type='log'))
    fig = go.Figure(dict(data=[fig_data], layout=layout), )
    iplot(fig)

    # Users tweeting a lot !
    n_tweeters = data.loc[selector, 'author_id'].nunique()
    print("\n\nNumber of tweeters on {} : {}".format(topic, n_tweeters))

    print("\nUsers most tweeting")
    n_tweets_by_user = data.loc[selector, 'author_id'].value_counts()
    n_tweets_by_user = pd.DataFrame(n_tweets_by_user).reset_index()
    n_tweets_by_user.columns = ['user_id', '# of Tweets']
    return n_tweets_by_user.head(10)
